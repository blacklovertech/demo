import numpy as np
import pandas as pd
from sklearn.cluster import DBSCAN
from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt

data = pd.read_csv("ccgeneral.csv")
data.head()

data.isnull().sum()

data = data.drop('CUST_ID', axis=1)
data.fillna(data.mean(), inplace=True)

scaler = StandardScaler()
x_scaled = scaler.fit_transform(data)

pca = PCA(n_components=2)
x_principal = pca.fit_transform(x_scaled)
x_principal = pd.DataFrame(x_principal, columns=['V1', 'V2'])

print(x_principal.head())

dbscan = DBSCAN(eps=0.036, min_samples=4).fit(x_principal)
labels = dbscan.labels_

data['cluster'] = labels
print(data.tail())

clusterColor = {0: 'yellow', 1: 'green', 2: 'blue', -1: 'red'}
colors = [clusterColor.get(label, 'black') for label in labels]

plt.figure(figsize=(12, 10))
plt.scatter(x_principal['V1'], x_principal['V2'], c=colors)
plt.title("Implementation of DBSCAN Clustering", fontname="Arial", fontweight="bold")
plt.show()
